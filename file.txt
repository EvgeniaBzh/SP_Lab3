vector<int> readArrayFromFile(const string& filePath) {
    vector<int> arr;
    ifstream inFile(filePath);
    int number;

    if (inFile.is_open()) {
        while (inFile >> number) {  
            arr.push_back(number);
        }
        inFile.close();
        cout << "Array read from " << filePath << endl;
    }
    else {
        cout << "Unable to open file " << filePath << endl;
    }

    return arr;
}

void writeArrayToFile(const vector<int>& arr, const string& filePath) {
    ofstream outFile(filePath);

    if (outFile.is_open()) {
        for (size_t i = 0; i < arr.size(); ++i) {
            outFile << arr[i] << " "; 
            if ((i + 1) % 10 == 0) {
                outFile << endl; 
            }
        }
        outFile.close();
        cout << "Sorted array written to " << filePath << endl;
    }
    else {
        cout << "Unable to open file " << filePath << endl;
    }
}
